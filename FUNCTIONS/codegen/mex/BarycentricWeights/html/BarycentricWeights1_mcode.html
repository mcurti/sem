1function [ w ] = BarycentricWeights( x )
 2%BarycentricWeights Is returning the weights for Lagrange interpolation
 3%algorithm 30
 4%   For a set of points "x_i" on x axis the functions returns the weights
 5%   for Lagrange interpolation
 6
 7N = length(x);
 8
 9w = ones(1,N);
10
11for jj = 2:(N)
12    for k = 1:(jj-1)
13           w(k)  = w(k) *(x(k ) - x(jj));
14           w(jj) = w(jj)*(x(jj) - x(k ));
15    end
16end
17
18w = 1./w;
